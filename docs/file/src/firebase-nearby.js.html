export default class FirebaseNearby {

    this._db = new FirebaseNearbyDb(ref, this._defaultConfig);

  /**
   * Returns all the locations within the provided Boundaries (rectangle / square only)
   * 
   * @param {Object} entry the new entry
   * @param {number} minPrecision holds percision for the geohash e.g. 6 characters long
   * @param {number} maxPrecision holds percision for the geohash e.g. 6 characters long
   * @param {number} startAt the number of first entry
   * @param {number} endAt the number of the last entry
   */
  getLocationsByBoundaries(boundaries, precision, startAt = 1, endAt = 50) {
    const geohashesInBoundaries = this._utils.getGeohashesInBoundaries(boundaries, precision); // geohash strings within the given boundaries
    
    return this._db.getLocationsFromGeohashes(geohashesInBoundaries).then(geohashNode => {
      return this._utils.getLocationsInBoundaries(geohashNode, boundaries);
    }).then(pushKeys => {
      return this._db.fetchLocations(pushKeys, startAt,  endAt);
    });
  }

  /**
   * Creates a new entry in the database
   *
   * @param {Object} entry the new entry
   * @param {number} minPrecision holds percision for the geohash e.g. 6 characters long
   * @param {number} maxPrecision holds percision for the geohash e.g. 6 characters long
   */
  createEntry(entry, minPrecision = this._defaultConfig.minPrecision, maxPrecision = this._defaultConfig.maxPrecision) {
    const geohash = this._utils.getGeohashByLocation(entry.location, maxPrecision);
    const geohashPrecisions = this._utils.getGeohashPrecisions(geohash, minPrecision, maxPrecision);

    this._db.pushEntry(entry).then(data => {
      this._db.setGeohashLocations(data.key, entry.location, geohashPrecisions);
    });
  }
